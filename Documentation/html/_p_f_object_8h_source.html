36  // An array of NSDictionary of NSString -> PFFieldOperation.
37  // Each dictionary has a subset of the object's keys as keys, and the
38  // changes to the value for that key as its value.
39  // There is always at least one dictionary of pending operations.
40  // Every time a save is started, a new dictionary is added to the end.
41  // Whenever a save completes, the new data is put into fetchedData, and
42  // a dictionary is removed from the start.
43  NSMutableArray *PF_NULLABLE_S operationSetQueue;
44 
45  // Our best estimate as to what the current data is, based on
46  // the last fetch from the server, and the set of pending operations.
47  NSMutableDictionary *PF_NULLABLE_S estimatedData;
48 }
